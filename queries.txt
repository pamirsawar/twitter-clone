CREATE TABLE `twitter`.`users` (
  `uid` INT NOT NULL,
  `username` VARCHAR(45) NOT NULL,
  `useremail` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  `dob` DATE NOT NULL,
  `firstname` VARCHAR(45) NOT NULL,
  `lastname` VARCHAR(45) NULL,
  PRIMARY KEY (`uid`),
  UNIQUE INDEX `uid_UNIQUE` (`uid` ASC) VISIBLE,
  UNIQUE INDEX `username_UNIQUE` (`username` ASC) VISIBLE,
  UNIQUE INDEX `useremail_UNIQUE` (`useremail` ASC) VISIBLE);




CREATE TABLE `twitter`.`tweets` (
  `tid` INT NOT NULL,
  `uid` INT NOT NULL,
  `content` VARCHAR(45) NOT NULL,
  `likecnt` VARCHAR(45) NULL,
  `retweetcnt` VARCHAR(45) NULL,
  PRIMARY KEY (`tid`),
  UNIQUE INDEX `tid_UNIQUE` (`tid` ASC) VISIBLE,
  INDEX `uid_idx` (`uid` ASC) VISIBLE,
  CONSTRAINT `uid`
    FOREIGN KEY (`uid`)
    REFERENCES `twitter`.`users` (`uid`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
    
    
    
    CREATE TABLE `twitter`.`follwers` (
  `fid` INT NOT NULL,
  `uid` INT NULL,
  `following` INT NULL,
  PRIMARY KEY (`fid`),
  INDEX `following_idx` (`following` ASC) VISIBLE,
  CONSTRAINT `following`
    FOREIGN KEY (`following`)
    REFERENCES `twitter`.`users` (`uid`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);


CREATE TABLE `twitter`.`comments` (
  `cid` INT NOT NULL AUTO_INCREMENT,
  `content` VARCHAR(100) NULL,
  `username` VARCHAR(45) NULL,
  `timestamp` VARCHAR(45) NULL,
  `tid` INT NULL,
  PRIMARY KEY (`cid`),
  INDEX `tid_idx` (`tid` ASC) VISIBLE,
  INDEX `username_idx` (`username` ASC) VISIBLE,
  CONSTRAINT `username`
    FOREIGN KEY (`username`)
    REFERENCES `twitter`.`users` (`username`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `tid`
    FOREIGN KEY (`tid`)
    REFERENCES `twitter`.`tweets` (`tid`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
